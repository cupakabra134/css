<?php
if ($_SERVER['REQUEST_METHOD'] === 'POST') {
    $message = htmlspecialchars($_POST['message']); // Zabezpečení
    $file = 'messages.txt';
    file_put_contents($file, $message . "\n", FILE_APPEND);
    echo "Zpráva byla uložena. <a href='index.html'>Zpět</a>";
} else {
    header('Location: index.html');
}
?>
